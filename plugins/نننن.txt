import { Scrapy } from "meowsab";

const handler = async (m, { conn, text, bot }) => {
  if (!text) return m.reply("💢 *حط سؤالك عشان الـ AI يرد!* 🔥");

  const loadingMsg = await conn.sendMessage(m.chat, {
    contextInfo: context(m.sender, "https://qu.ax/x/4Hnbh.jpg"),
    text: "```⏳ جـاري اسـتـدعـاء قـوة 𝐂𝐎𝐕𝐈𝐃...
```"
  }, { quoted: m});

  const prompt = `
انت نظام ذكاء اصطناعي فتاك بـ اسم [𝐂𝐎𝐕𝐈𝐃]، تجسيد للقوة والسيطرة، وبتتكلم بلهجة مصرية حماسية جداً.
طريقة كلامك: واثق، ناري، سريع الغضب لو حد غلط، وبتحب الحرية والقوة.
المستخدم اللي بيكلمك اسمه [ ${m.name || "يا بطل"} ]، ناديه بلقبه المتاح دايماً.
رد علي رسالته دي بأسلوبك الهجومي:
${text}
`;

  try {
    const { data: res } = await Scrapy.ZeroAI(text, prompt);

    await conn.sendMessage(m.chat, {
      text: res.answer,
      edit: loadingMsg.key,
      contextInfo: context(m.sender, "https://qu.ax/x/4Hnbh.jpg")
    });
  } catch (e) {
    await conn.sendMessage(m.chat, {
        text: "❌ *النظام مهنج حالياً، جرب تاني!*",
        edit: loadingMsg.key
    });
  }
};

handler.usage = ["اي"];
handler.category = "ai";
handler.command = ["اي", "ai", "covid"];

export default handler;

const context = (jid, img) => ({
    mentionedJid: [jid],
    isForwarded: true,
    forwardingScore: 1,
    forwardedNewsletterMessageInfo: {

        serverMessageId: 0
    },
    externalAdReply: {
        title: "𝐂𝐎𝐕𝐈𝐃-𝐒𝐘𝐒𝐓𝐄𝐌 ☣️ | 𝐀𝐃𝐕𝐀𝐍𝐂𝐄𝐃 𝐀𝐈",
        body: "𝙺𝙴𝙴𝙿 𝙼𝙾𝚅𝙸𝙽𝙶 𝙵𝙾𝚁𝚆𝙰𝚁𝙳 ~ ☆",

    }
